function calculateScore() {
let score = 0;
let inputs = document.querySelectorAll('input[type="radio"]:checked');
inputs.forEach(input => {
score += parseInt(input.value);
});
let resultText = "";
if (score >= 45) {
resultText = "Pola Pikir Berkembang yang Kuat";
} else if (score >= 34) {
resultText = "Pola Pikir Berkembang dengan Beberapa Pola Pikir Tetap";
} else if (score >= 21) {
resultText = "Pola Pikir Tetap dengan Beberapa Pola Pikir Berkembang";
} else {
resultText = "Pola Pikir Tetap yang Kuat";
}
document.getElementById('result').innerText = "Skor Anda: " + score + " - " + resultText;
}
</script>
